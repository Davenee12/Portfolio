FTIC COHORT RETENTION ANALYSIS

Tracking First-Time-in-College (FTIC) student retention is critical for university accreditation and strategic planning. I developed a comprehensive SQL view that tracks cohorts across multiple years, calculating 1-year through 4-year retention rates that feed directly into executive dashboards.

The query uses CTEs to first identify students in federal cohort groups (Full-time Fall and Part-time Fall entries), then tracks their return status across subsequent fall terms. It handles complex business logic including active credit verification and enrollment status checks, outputting student-level data with demographic breakdowns for IPEDS federal reporting.

-- FTIC Cohort Retention Analysis View
-- Tracks student retention across 4 years by federal cohort

WITH cohort_base AS (
  SELECT DISTINCT
    ost.sttr_student AS studentid,
    LEFT(sal.sta_fed_cohort_group, 2)::integer AS cohortterm,
    pnc.gender,
    xie.ipeds_ethnicity AS raceethnicity,
    pnc.first_name AS firstname,
    pnc.last_name AS lastname
  FROM dbo.ods_student_terms ost
  JOIN dbo.spt_student_acad_levels sal 
    ON sal.sta_student = ost.sttr_student
  JOIN dbo.spt_person_non_corp pnc 
    ON pnc.id = ost.sttr_student
  JOIN dbo.x_ipeds_ethnicity xie 
    ON xie.id = ost.sttr_student
  WHERE sal.sta_fed_cohort_group IN (
    '21FF', '21FP', '22FF', '22FP',
    '23FF', '23FP', '24FF', '24FP'
  )
),

returns_by_term AS (
  SELECT sttr_student, sttr_term
  FROM dbo.ods_student_terms
  WHERE sttr_term IN (
    '2022FA', '2023FA', '2024FA', '2025FA'
  )
  AND sttr_current_status IN ('R', 'T')
  AND x_sttr_active_cred > 0
  GROUP BY sttr_student, sttr_term
)

SELECT
  c.studentid,
  c.cohortterm,
  CONCAT('20', c.cohortterm, '-', '20', c.cohortterm + 1) 
    AS academicyear,
  CONCAT('Fall 20', c.cohortterm) AS termdescription,
  c.gender,
  c.raceethnicity,
  c.firstname,
  c.lastname,
  -- 1-Year Retention
  CASE c.cohortterm
    WHEN 21 THEN CASE WHEN r22.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
    WHEN 22 THEN CASE WHEN r23.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
    WHEN 23 THEN CASE WHEN r24.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
    WHEN 24 THEN CASE WHEN r25.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
  END AS oneyearretention,
  -- 2-Year Retention
  CASE c.cohortterm
    WHEN 21 THEN CASE WHEN r23.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
    WHEN 22 THEN CASE WHEN r24.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
    WHEN 23 THEN CASE WHEN r25.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
  END AS twoyearretention,
  -- 3-Year Retention
  CASE c.cohortterm
    WHEN 21 THEN CASE WHEN r24.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
    WHEN 22 THEN CASE WHEN r25.sttr_student IS NOT NULL 
      THEN 1 ELSE 0 END
  END AS threeyearretention,
  -- 4-Year Retention
  CASE WHEN c.cohortterm = 21 
    AND r25.sttr_student IS NOT NULL
    THEN 1 ELSE 0 
  END AS fouryearretention
FROM cohort_base c
LEFT JOIN returns_by_term r22 
  ON r22.sttr_student = c.studentid 
  AND r22.sttr_term = '2022FA'
LEFT JOIN returns_by_term r23 
  ON r23.sttr_student = c.studentid 
  AND r23.sttr_term = '2023FA'
LEFT JOIN returns_by_term r24 
  ON r24.sttr_student = c.studentid 
  AND r24.sttr_term = '2024FA'
LEFT JOIN returns_by_term r25 
  ON r25.sttr_student = c.studentid 
  AND r25.sttr_term = '2025FA'
ORDER BY c.cohortterm, c.lastname, c.firstname;

This comprehensive HR analytics view pulls from multiple source tables to create a unified employee roster with demographics, compensation, and organizational data. It powers our Staff Demographics dashboard and supports workforce planning initiatives.

-- Current Employees Roster View
-- Comprehensive HR data with salary, demographics, and org structure

WITH base AS (
  SELECT
    BTRIM(sp.ppwg_hrp_id::text) AS key_hrp,
    sp.ppwg_base_et,
    sp.ppwg_start_date,
    sp.ppwg_end_date,
    sp.ppwg_annualized_amt::numeric AS annualized_amt
  FROM dbo.spt_perposwg sp
  WHERE sp.ppwg_base_et::text = ANY (
    ARRAY['REGU'::text, 'PARS'::text]
  )
),

ranked_with_salary AS (
  SELECT
    b.key_hrp,
    b.ppwg_base_et,
    b.ppwg_start_date,
    b.ppwg_end_date,
    b.annualized_amt,
    ROW_NUMBER() OVER (
      PARTITION BY b.key_hrp
      ORDER BY 
        (b.ppwg_end_date IS NULL) DESC,
        b.ppwg_start_date DESC,
        b.annualized_amt DESC
    ) AS rn
  FROM base b
  WHERE b.annualized_amt IS NOT NULL 
    AND b.annualized_amt > 0::numeric
),

pay_choice AS (
  SELECT
    rws.key_hrp,
    rws.annualized_amt AS salary,
    rws.ppwg_start_date,
    rws.ppwg_end_date,
    CASE
      WHEN rws.ppwg_end_date IS NULL 
        THEN 'open_position'::text
      ELSE 'latest_with_salary'::text
    END AS salary_source
  FROM ranked_with_salary rws
  WHERE rws.rn = 1
),

pay_choice_normalized AS (
  SELECT
    pc.key_hrp,
    pc.salary,
    pc.ppwg_start_date,
    pc.ppwg_end_date,
    pc.salary_source,
    LTRIM(pc.key_hrp, '0'::text) AS key_hrp_no_zeros
  FROM pay_choice pc
),

ranked_data AS (
  SELECT
    hrper.hrper_id,
    person.last_name,
    person.first_name,
    person.birth_date,
    hrper.hrp_effect_employ_date AS hired_date,
    hrper.tenure_type_desc,
    hrper.position_desc,
    -- Position standardization (80+ mappings)
    CASE
      WHEN hrper.position_desc::text ~~* '%President%'::text
        AND hrper.position_desc::text !~~* '%VP%'::text
        AND hrper.position_desc::text !~~* '%Vice%'::text
        THEN 'President'::text
      WHEN hrper.position_desc::text ~~* '%VP for Financial Admin%'::text
        THEN 'Vice President for Financial Administration'::text
      WHEN hrper.position_desc::text ~* '^Prof[,.]?\s*(Bus|Business)\s*Admin'::text
        THEN 'Professor of Business Administration'::text
      -- ... 75+ additional CASE conditions ...
      ELSE INITCAP(hrper.position_desc::text)
    END AS position_desc_standardized,
    hrper.status_desc,
    CASE person.gender
      WHEN 'M'::text THEN 'Male'::text
      WHEN 'F'::text THEN 'Female'::text
      ELSE 'Not Specified'::text
    END AS gender,
    person.age,
    CASE
      WHEN person.age IS NULL OR person.age = 0 
        THEN 'Unknown'::text
      WHEN person.age >= 79 THEN 'Silent Generation'::text
      WHEN person.age >= 60 THEN 'Baby Boomer'::text
      WHEN person.age >= 44 THEN 'Generation X'::text
      WHEN person.age >= 28 THEN 'Millennial'::text
      WHEN person.age >= 12 THEN 'Generation Z'::text
      ELSE 'Generation Alpha'::text
    END AS generation_label,
    INITCAP(person.ipeds_race_ethnic_desc) AS ipeds_ethnicity,
    CASE
      WHEN hrper.position_desc::text ~~* '%Adjunct%'::text 
        THEN 'Faculty'::text
      WHEN hrper.position_desc::text ~~* '%Prof%'::text 
        THEN 'Faculty'::text
      ELSE 'Staff'::text
    END AS position_type,
    CASE
      WHEN hrper.position_desc::text ~* '(Assistant|Asst\.?)\s*Prof'::text
        THEN 'Assistant Professor'::text
      WHEN hrper.position_desc::text ~* '(Associate|Assoc\.?)\s*Prof'::text
        THEN 'Associate Professor'::text
      WHEN hrper.position_desc::text ~~* '%Professor%'::text
        THEN 'Professor'::text
      ELSE NULL::text
    END AS faculty_rank,
    d.depts_desc,
    d.division_desc,
    perstat.perstat_former_service_years,
    hrper.hrp_service_years,
    COALESCE(perstat.perstat_former_service_years, 0::numeric) 
      + COALESCE(hrper.hrp_service_years, 0::numeric) 
      AS total_service_years,
    pc.salary,
    pc.salary_source,
    ROW_NUMBER() OVER (
      PARTITION BY hrper.hrper_id
      ORDER BY
        CASE WHEN perstat.perstat_former_service_years IS NOT NULL
          THEN 0 ELSE 1 END,
        perstat.perstat_former_service_years DESC NULLS LAST
    ) AS rn
  FROM dbo.ods_hrper hrper
  JOIN dbo.ods_person person 
    ON hrper.hrper_id::text = person.id::text
  LEFT JOIN dbo.ods_depts d 
    ON d.depts_id::text = hrper.hrp_pri_dept_sort::text
  LEFT JOIN dbo.spt_perstat perstat 
    ON hrper.hrper_id::text = perstat.perstat_hrp_id::text
  LEFT JOIN pay_choice_normalized pc 
    ON pc.key_hrp = hrper.hrper_id::text
    OR pc.key_hrp_no_zeros = hrper.hrper_id::text
  WHERE (hrper.hrp_current_status::text <> ALL (
    ARRAY['ST', 'TE', 'TM']::text[]
  ))
  AND (hrper.hrp_effect_term_date > CURRENT_DATE 
    OR hrper.hrp_effect_term_date IS NULL)
)

SELECT
  hrper_id, last_name, first_name, birth_date,
  hired_date, gender, tenure_type_desc, age,
  generation_label, ipeds_ethnicity, position_desc,
  position_type, faculty_rank, status_desc,
  perstat_former_service_years, hrp_service_years,
  depts_desc, division_desc, total_service_years,
  salary, salary_source
FROM ranked_data
WHERE rn = 1
ORDER BY last_name, first_name;
